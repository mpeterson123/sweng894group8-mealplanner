$PHPUnitOFF = $_SERVER['phpunit_off'] ?? FALSE;

function PHPUnitCheck($errors)
{
    global $PHPUnitOFF;

    if ($PHPUnitOFF)
    {
        return;
    }

    if (count($errors) > 0)
    {
        throw new Exception('Errors were generated.');
    }
}
